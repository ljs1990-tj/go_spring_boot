# 🧑‍🍳 2단계: 백엔드 3형제와 비밀 레시피 (통합 코드 모음)

아래의 5개 파일은 사용자가 "게시판 목록 보여주세요!"라고 요청했을 때, DB에서 데이터를 꺼내서 전달하는 하나의 완벽한 세트입니다.

---

== BoardDTO.java ==
```java
// [도시락통] 데이터를 안전하게 담아서 이동시키는 쟁반 역할입니다.
package com.example.myproject.dto;

import lombok.Data;

// @Data: 롬복(Lombok)의 마법입니다! 이 명찰 하나만 붙이면
// 뚜껑을 열고 닫는 귀찮은 코드들(Getter, Setter, toString 등)을 스프링이 알아서 만들어줍니다.
@Data
public class BoardDTO {
    // 게시판 테이블(DB)의 컬럼들과 똑같은 이름으로 변수를 만듭니다.
    private int id;          // 글 번호
    private String title;    // 글 제목
    private String content;  // 글 내용
    private String writer;   // 작성자 이름
}
```

== BoardMapper.java ==
```java
// [주방 조수] 주방장(Service)의 지시를 받는 자바 인터페이스(설계도)입니다.
package com.example.myproject.mapper;

import java.util.List;
import org.apache.ibatis.annotations.Mapper;
import com.example.myproject.dto.BoardDTO;

// @Mapper: 스프링에게 "이 친구가 DB 냉장고 문을 여는 조수(마이바티스)야!" 라고 알려줍니다.
@Mapper
public interface BoardMapper {
    
    // 주방장이 내리는 지시사항입니다. 
    // "게시판 목록(BoardDTO)을 여러 개(List) 담아서 가져와!" 라는 뜻이죠.
    // 주의: 이 함수의 이름('getBoardList')은 잠시 후 작성할 XML 파일의 id와 토시 하나 안 틀리고 똑같아야 합니다!
    List<BoardDTO> getBoardList();
}
```

---

== BoardMapper.xml ==
```xml
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="com.example.myproject.mapper.BoardMapper">

    <select id="getBoardList" resultType="com.example.myproject.dto.BoardDTO">
        /* 게시판 테이블에서 글 번호, 제목, 내용, 작성자를 최신순(내림차순)으로 가져와라! */
        SELECT 
            id, 
            title, 
            content, 
            writer 
        FROM board 
        ORDER BY id DESC
    </select>

</mapper>
```

== BoardService.java ==
```java
// [메인 주방장] 식당의 핵심 로직(요리)을 담당하는 곳입니다.
package com.example.myproject.service;

import java.util.List;
import org.springframework.stereotype.Service;
import com.example.myproject.mapper.BoardMapper;
import com.example.myproject.dto.BoardDTO;

// @Service: "이 클래스가 주방장이야!" 라고 알려주는 명찰입니다.
@Service
public class BoardService {

    // 주방장은 요리를 위해 조수(Mapper)가 반드시 필요합니다.
    private final BoardMapper boardMapper;

    // 생성자를 통해 스프링이 조수를 알아서 주방장 옆에 데려다 놓습니다. (이를 의존성 주입-DI 라고 부릅니다)
    public BoardService(BoardMapper boardMapper) {
        this.boardMapper = boardMapper;
    }

    // 매니저(Controller)가 "게시판 목록 요리해주세요!" 라고 주문을 넣으면 실행되는 함수입니다.
    public List<BoardDTO> getBoardList() {
        
        // 지금은 단순 조회라서 바로 조수에게 지시하지만,
        // 만약 "VIP 회원만 볼 수 있게 해라" 같은 복잡한 조건(if문)이 있다면 무조건 여기서 처리해야 합니다!
        System.out.println("주방장: 조수야 DB에서 게시판 목록 좀 꺼내와!");
        
        return boardMapper.getBoardList(); // 조수가 가져온 데이터를 그대로 매니저에게 전달
    }
}
```

== BoardController.java ==
```java
// [홀 매니저 - JSON 전용] 손님(Vue.js)의 Ajax 요청을 받고 데이터를 서빙합니다.
package com.example.myproject.controller;

import java.util.List;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RestController;
import com.example.myproject.service.BoardService;
import com.example.myproject.dto.BoardDTO;

// @RestController: 1단계의 @Controller와 다르게 'Rest'가 붙었습니다.
// JSP 화면(메뉴판)을 보여주는 게 아니라, "데이터(JSON)"만 깔끔하게 포장해서 손님에게 던져주는 특별한 매니저입니다.
// Vue.js와 통신할 때는 무조건 이 명찰을 달아야 합니다!
@RestController
public class BoardController {

    // 매니저도 주문을 전달할 주방장(Service)이 필요합니다.
    private final BoardService boardService;

    // 스프링이 알아서 주방장을 매니저 옆에 데려다 놓습니다.
    public BoardController(BoardService boardService) {
        this.boardService = boardService;
    }

    // @PostMapping: 손님(Vue.js의 $.ajax)이 POST 방식으로 아래 주소로 찾아왔을 때 실행됩니다.
    // 1단계 index.jsp의 $.ajax { url: "/api/board/list" } 와 똑같이 맞춰주어야 합니다.
    @PostMapping("/api/board/list")
    public List<BoardDTO> getBoardList() {
        
        System.out.println("매니저: /api/board/list 요청 들어옴! 주방장에게 주문 전달!");
        
        // 주방장이 완성해 준 요리(데이터 목록)를 받아서 리턴합니다.
        // 그러면 스프링이 알아서 이 자바 리스트를 JSON 형태(대괄호와 중괄호 모양)로 예쁘게 변환해서 Vue로 쏴줍니다!
        return boardService.getBoardList(); 
    }
}
```
