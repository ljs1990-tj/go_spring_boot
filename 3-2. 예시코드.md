# 🍽️ 3단계: 화려한 홀 서빙 기술 (프론트엔드 통합 코드) 

1단계에서 만들었던 `index.jsp` 파일의 내용을 아래 코드로 통째로 바꿔주세요. 2단계의 백엔드(주방)가 정상적으로 켜져 있다면, 화면이 켜지자마자 DB에 있는 게시판 목록이 예쁘게 출력될 것입니다!

---

== index.jsp ==
```jsp
<%@ page language="java" contentType="text/html; charset=UTF-8" pageEncoding="UTF-8"%>
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>웹 식당 게시판</title>
    <script src="[https://code.jquery.com/jquery-3.7.1.js](https://code.jquery.com/jquery-3.7.1.js)" integrity="sha256-eKhayi8LEQwp4NKxN+CfCh+3qOVUtJn3QNZ0TciWLP4=" crossorigin="anonymous"></script>
    <script src="[https://unpkg.com/vue@3/dist/vue.global.js](https://unpkg.com/vue@3/dist/vue.global.js)"></script>
    <style>
        /* 손님들이 보기 편하도록 메뉴판(테이블)을 꾸며줍니다. */
        table, tr, td, th {
            border : 1px solid black;
            border-collapse: collapse;
            padding : 10px 20px;
            text-align: center;
        }
        th { background-color: #f2f2f2; }
        tr:nth-child(even) { background-color: #fafafa; }
        .btn-style { margin-bottom: 10px; padding: 5px 10px; cursor: pointer; }
    </style>
</head>
<body>
    <div id="app">
        <h2>📋 우리 식당 자유게시판</h2>
        
        <button class="btn-style" @click="fnList">최신 목록 불러오기 🔄</button>

        <table>
            <thead>
                <tr>
                    <th>글 번호</th>
                    <th>제 목</th>
                    <th>작성자</th>
                </tr>
            </thead>
            <tbody>
                <tr v-for="item in boardList" :key="item.id">
                    <td>{{ item.id }}</td>
                    <td>{{ item.title }}</td>
                    <td>{{ item.writer }}</td>
                </tr>
                
                <tr v-if="boardList.length === 0">
                    <td colspan="3">아직 작성된 글이 없습니다. 첫 글을 남겨주세요!</td>
                </tr>
            </tbody>
        </table>
    </div>
</body>
</html>

<script>
    // 최첨단 지배인(Vue)의 뇌 구조 세팅
    const app = Vue.createApp({
        // 1구역: 지배인의 전용 메모장 (반응성 데이터)
        data() {
            return {
                // 주방에서 받아올 게시글 목록을 담기 위해 '빈 배열(접시)'을 준비해 둡니다.
                boardList: []
            };
        },
        
        // 2구역: 지배인의 행동 지침서 (함수 모음)
        methods: {
            fnList: function () {
                // 🚨 [초보자 생명줄] 🚨
                // Ajax 안으로 들어가면 'this'가 지배인에서 점원으로 바뀌어버립니다!
                // 그래서 출발하기 전에 진짜 지배인(this)을 'self'라는 안전한 상자에 복사해 둡니다.
                let self = this; 
                
                // 주방에 전달할 특별 요청사항이 있다면 여기에 담습니다. (지금은 비워둡니다)
                let param = {}; 
                
                // 투명 인간 점원(Ajax) 출발!
                $.ajax({
                    url: "/api/board/list", // 목적지: 2단계 BoardController의 @PostMapping 주소
                    dataType: "json",       // "요리는 무조건 JSON 객체로 가져와!"
                    type: "POST",           // POST 방식으로 이동
                    data: param,
                    
                    // 점원이 주방에서 무사히 요리(data)를 받아왔을 때 실행됩니다!
                    success: function (data) {
                        // 요리가 잘 나왔는지 개발자 도구(F12) 콘솔 창에서 확인합니다. (디버깅 필수)
                        console.log("주방에서 나온 요리: ", data); 
                        
                        // ★ 마법 발동 ★
                        // 점원이 가져온 요리(data)를 아까 챙겨둔 진짜 지배인(self)의 메모장에 쏟아붓습니다.
                        // 이 순간, 화면의 테이블이 알아서 스르륵 업데이트됩니다!
                        self.boardList = data; 
                    },
                    
                    // 길을 잃거나(404) 주방에서 불이 났을 때(500) 실행됩니다.
                    error: function (request, status, error) {
                        alert("데이터를 불러오는데 실패했습니다! F12를 눌러 콘솔을 확인하세요.");
                        console.log("에러 원인: ", error);
                    }
                });
            }
        }, 
        
        // 3구역: 식당 오픈 직전 최종 점검
        mounted() {
            let self = this;
            // 손님이 이 웹페이지에 들어오자마자, 자동으로 "최신 목록 불러오기" 행동(fnList)을 1회 실행합니다.
            // 덕분에 손님은 빈 화면을 보지 않고 바로 게시판 목록을 볼 수 있습니다.
            self.fnList();
        }
    });

    // 세팅이 끝난 지배인을 홀(id="app" 인 HTML 영역)에 투입시킵니다!
    app.mount('#app');
</script>
```
